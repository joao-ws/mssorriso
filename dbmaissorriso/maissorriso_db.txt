CREATE DATABASE maissorriso_db
	DEFAULT CHARACTER SET utf8
	COLLATE utf8_general_ci;

USE maissorriso_db;

CREATE TABLE paciente (

	idpaciente INT AUTO_INCREMENT PRIMARY KEY,

	nome_paciente VARCHAR(45) NOT NULL UNIQUE,
	cpf bigint(11) UNSIGNED ZEROFILL NOT NULL UNIQUE

);

CREATE TABLE profissional (

	idodonto INT AUTO_INCREMENT PRIMARY KEY,

	nome_profissional VARCHAR(45) NOT NULL UNIQUE,
	cro bigint(20) UNSGINED ZEROFILL NOT NULL UNIQUE

);

CREATE TABLE consulta (

	idconsulta INT AUTO_INCREMENT PRIMAY KEY,
	idpaciente INT NOT NULL,
	idodonto INT NOT NULL,

	hora TIME AUTO_INCREMENT NOT NULL UNIQUE,
	data DATE AUTO_INCREMENT NOT NULL UNIQUE,

	FOREIGN KEY (idpaciente) REFERENCES paciente(idpaciente)
	FOREIGN KEY (idodonto)	REFERENCES idodonto(profissional)
);
